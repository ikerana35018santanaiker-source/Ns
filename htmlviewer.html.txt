    // Simula la carga
    setTimeout(() => {
      document.getElementById("loadingScreen").style.display = "none";

    function setUI(type) {
      document.getElementById("mainMenu").style.display = "none";
      document.getElementById("dashboard").style.display = "flex";
      document.getElementById("sideButtons").style.display = "flex";
      document.body.className = type === "pc" ? "pcUI" : "mobileUI";
    }

    // Manejo del icono de usuario
    const userIcon = document.getElementById("userIcon");
    const uploadIcon = document.getElementById("uploadIcon");
    const savedIcon = localStorage.getItem("userIcon");
    if (savedIcon) userIcon.src = savedIcon;
    userIcon.addEventListener("click", () => uploadIcon.click());
    uploadIcon.addEventListener("change", (event) => {
      const file = event.target.files[0];
      if (file) {
        const reader = new FileReader();
        reader.onload = e => {
          userIcon.src = e.target.result;
          localStorage.setItem("userIcon", e.target.result);
        };
        reader.readAsDataURL(file);
      }
    });

    // Overlay bÃºsqueda
    const searchOverlay = document.getElementById("searchOverlay");
    document.getElementById("searchBtn").addEventListener("click", () => {
      searchOverlay.classList.add("active");

    });
    document.getElementById("closeSearch").addEventListener("click", () => {
      searchOverlay.classList.remove("active");
    });

    // Popup juegos
    const gamePopup = document.getElementById("gamePopup");
    const gameThumbnail = document.getElementById("gameThumbnail");
    const gameTitle = document.getElementById("gameTitle");
    const gameLink = document.getElementById("gameLink");
    document.getElementById("closePopup").addEventListener("click", () => gamePopup.style.display = "none");

    const juegos = {

    };

    document.getElementById("searchInput").addEventListener("keydown", (e) => {
      if (e.key === "Enter") {
        const query = e.target.value.trim();
        if (juegos[query]) {
          const juego = juegos[query];
          gameThumbnail.src = juego.thumbnail;
          gameTitle.textContent = query;
          gameLink.onclick = () => window.open(juego.link, "_blank");
          gamePopup.style.display = "flex";
        } else {
          alert("No se encontrÃ³ el juego. Usa un nombre exacto.");
        }
      }
